/**

возвращает значение площади треугольника, которое высчитывается по формуле Герона
@param {storona} a - Первый катет
@param {storona} b - Второй катет
@param {storona} c - Гипотенуза
@param {znachenie} p - Значение полупериметра треугольника

*/
function Treg(a = 3,b = 4,c = 5) {
  p = (a+b+c)/2;
  return Math.sqrt(p*(p-a)*(p-b)*(p-c));
}
console.log(Treg(3,4,5))
